-- 6. LOGIC ĐỊNH VỊ TRÁI (Fruit ESP)
spawn(function()
    while task.wait(1) do -- Quét mỗi giây 1 lần để tránh lag
        -- Xóa các ESP cũ
        for _, v in pairs(game.CoreGui:GetChildren()) do
            if v.Name == "FruitTrack" then v:Destroy() end
        end
        
        -- Nếu bật ESP
        if _G.ESP then
            for _, v in pairs(workspace:GetChildren()) do
                -- Tìm các vật phẩm chứa chữ "Fruit"
                if v.Name:find("Fruit") or (v:IsA("Tool") and v:FindFirstChild("Handle")) then
                    local bill = Instance.new("BillboardGui", game.CoreGui)
                    bill.Name = "FruitTrack"
                    bill.AlwaysOnTop = true
                    bill.Size = UDim2.new(0, 100, 0, 50)
                    bill.Adornee = v:IsA("Tool") and v.Handle or v
                    
                    local txt = Instance.new("TextLabel", bill)
                    txt.Size = UDim2.new(1, 0, 1, 0)
                    txt.BackgroundTransparency = 1
                    txt.Text = "🍎 " .. v.Name
                    txt.TextColor3 = Color3.fromRGB(255, 0, 0)
                    txt.Font = Enum.Font.GothamBold
                    txt.TextSize = 14
                end
            end
        end
    end
end)
